def execute(command)
  until accepted_any?
    Thread.pass
    sleep 0.01
  end

  json = command.to_json
  data = TEMPLATE % [json.length, json]

  @next_socket.write data
  @next_socket.close

  JSON.parse read_response(@queue.pop)
end
