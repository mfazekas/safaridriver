def self.launcher
  launcher =  case Platform.os
              when :windows
                WindowsLauncher.new
              when :macosx
                MacOSXLauncher.new
              when :unix, :linux
                UnixLauncher.new
              else
                raise "unknown OS: #{Platform.os}"
              end

  launcher
end
