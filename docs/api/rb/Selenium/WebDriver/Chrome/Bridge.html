def executeScript(script, *args)
  typed_args = args.map { |e| wrap_script_argument(e) }

  resp = execute :request => 'executeScript',
                 :script  => script,
                 :args    => typed_args

  return if resp.nil?
  unwrap_script_argument resp
end
