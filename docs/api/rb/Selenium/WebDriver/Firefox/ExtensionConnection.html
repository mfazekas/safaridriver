def connect(timeout = 20)
  Timeout.timeout(timeout) {
    loop do
      begin
        return TCPSocket.new(@host, @port).close
      rescue Errno::ECONNREFUSED, Errno::ENOTCONN, SocketError => e
        $stderr.puts "#{self} caught #{e.message} for #{@host}:#{@port}" if $DEBUG
        sleep 0.25
      end
    end
  }
end
