def start
  @pid = fork do
    unless $DEBUG
      [STDOUT, STDERR].each { |io| io.reopen("/dev/null") }
    end

    exec(*@args)
  end

  self
end

def ugly_death?
  code = exit_value()
  # if exit_val is nil, the process is still alive
  code && code != 0
end

def wait
  raise "called wait with no pid" unless @pid
  Process.waitpid2 @pid
rescue Errno::ECHILD
  nil
end
