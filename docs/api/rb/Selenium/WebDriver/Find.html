def find_element(*args)
  how, what = extract_args(args)

  unless by = FINDERS[how.to_sym]
    raise ArgumentError, "cannot find element by #{how.inspect}"
  end

  meth = "findElement#{by}"
  what = what.to_s

  bridge.send meth, ref, what
end
